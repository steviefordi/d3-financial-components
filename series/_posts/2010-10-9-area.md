example-code: |
  var area = fc.series.area()
      .xScale(xScale)
      .yScale(yScale);

  container.append('g')
      .datum(data)
      .call(area);

example-band-code: |
  var area = fc.series.area()
      .xScale(xScale)
      .yScale(yScale)
      .y1Value(function (d) { return d.high; } )
      .y0Value(function (d) { return d.low; } );

  container.append('g')
      .datum(data)
      .call(area);
